mybatis介绍:

前生是ibatis，它封装了JDBC的持久层框架，提供了对数据库操作的接口：SqlSessionFactoryBuilder SqlSessionFactory SqlSession  
它是半ORM框架（类和表的映射）

mybatis和Spring整合

步骤

1.导jar包：

spring-jdbc:4.2.4.RELEASE 
mysql-connector-java:5.1.32
mybatis:3.2.8
mybatis-spring:1.2.2

2.配置文件

<1> 在applicationContext-*.xml 文件中

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
xmlns:context="http://www.springframework.org/schema/context" xmlns:p="http://www.springframework.org/schema/p"
xmlns:aop="http://www.springframework.org/schema/aop" xmlns:tx="http://www.springframework.org/schema/tx"
xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans-4.2.xsd
http://www.springframework.org/schema/context http://www.springframework.org/schema/context/spring-context.xsd
http://www.springframework.org/schema/aop http://www.springframework.org/schema/aop/spring-aop.xsd 
http://www.springframework.org/schema/tx http://www.springframework.org/schema/tx/spring-tx.xsd
http://www.springframework.org/schema/util http://www.springframework.org/schema/util/spring-util.xsd">

   <!-- 数据库连接池 -->
   <!-- 加载配置文件 -->
     <context:property-placeholder location="classpath*:properties/*.properties" />
   <!-- 数据库连接池 jdbc dbcp 等类型-->
     <bean id="dataSource" class="org.springframework.jdbc.datasource.DriverManagerDataSource">
       <property name="url" value="${jdbc.url}" />
       <property name="username" value="${jdbc.username}" />
       <property name="password" value="${jdbc.password}" />
       <property name="driverClassName" value="${jdbc.driver}" />
      </bean>
    <!-- 让spring管理sqlsessionfactory 使用mybatis和spring整合包中的 -->
     <bean id="sqlSessionFactory" class="org.mybatis.spring.SqlSessionFactoryBean">
    <!-- 数据库连接池 -->
      <property name="dataSource" ref="dataSource" />
     <!-- 加载mybatis的全局配置文件 可省略 -->
     </bean>
     <!-- mapper扫描，扫描所有的mapper接口 -->
     <bean class="org.mybatis.spring.mapper.MapperScannerConfigurer">
      <property name="basePackage" value="com" />
     </bean>
</bean>

<2> 在 properties/*.properties

jdbc.driver=com.mysql.jdbc.Driver
jdbc.url=jdbc:mysql://localhost:3306/database
jdbc.username=root
jdbc.password=123456

3.建立映射接口和代理mapper

(1)接口：

public interface UserMapper {
 //查询所有，返回map类型
List<Map<String, Object>> findAll();
}

(2)*mapper.xml:

 <?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper    
PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"    
"http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.mapper.UserMapper">
 <select id="findAll" resultType="Map">
   select * from user
 </select>
</mapper>

4.测试

public class myBatisTest {
 ApplicationContext ctx;
  @Before
 public void setUp() throws Exception {
    //手动启动Spring容器
  ctx = new ClassPathXmlApplicationContext(
       "spring/applicationContext-mybatis.xml");
 }

   @Test
  public void test1(){
  /**
   * mybatis和hibernate都是持久层的框架（对jdbc的封装），都是ORM。
   * mybatis是半ORM，专注SQL的编写,对数据库的操作更灵活
   * hibernate是全ORM，不需要写SQL
   */
    UserMapper mapper=(UserMapper) ctx.getBean("userMapper");
    List<Map<String, Object>> user=mapper.findAll();
      for(Map<String, Object> u:user) {
         System.out.println(u);
      }
    }
 }

注意：1.mybatis映射是有规范的，接口的包名必须和代理mapper文件名一一对应 ，反之会找不到该映射文件

或者 在 SqlSessionFactory中增加属性指定映射文件的位置：<property name="mapperLocations"  value="classpath:mapper/*.xml"/>
    2.mapper映射文件中的Sql 的 #和$的区别#的值是“”为防止加SQL，$则反之。

mybatis的逆向工程（提高效率，自动生成mapper、接口、实体）

1.为不影响现有工程，新建一个工程。

2.manven或者倒入3个相关的jar包：
   mybatis:3.2.8
   Mybatis-generator-core:1.3.5
   Mysql-connector-java:5.1.43

3.创建配置文件 （generatorConfig.xml）
    <?xml version="1.0" encoding="UTF-8"?>
   <!DOCTYPE generatorConfiguration
     PUBLIC "-//mybatis.org//DTD MyBatis Generator Configuration 1.0//EN"
    "http://mybatis.org/dtd/mybatis-generator-config_1_0.dtd">

 <generatorConfiguration>
  <context id="testTables" targetRuntime="MyBatis3">
    <commentGenerator>
        <!-- 是否去除自动生成的注释 true：是 ： false:否 -->
        <property name="suppressAllComments" value="true" />
    </commentGenerator>
    <!--数据库连接的信息：驱动类、连接地址、用户名、密码 -->
    <jdbcConnection driverClass="com.mysql.jdbc.Driver"
        connectionURL="jdbc:mysql://localhost:3306/mynote? 
        characterEncoding=utf-8&amp;useSSL=true" userId="root"      
        password="123456" >
    </jdbcConnection>
    <!-- 默认false，把JDBC DECIMAL 和 NUMERIC 类型解析为 Integer，为 true时把JDBC DECIMAL 和 
        NUMERIC 类型解析为java.math.BigDecimal -->
    <javaTypeResolver>
        <property name="forceBigDecimals" value="false" />
    </javaTypeResolver>

    <!-- targetProject:生成PO类的位置 -->
    <javaModelGenerator targetPackage="cn.xixipro.pojo"
        targetProject="./src/main/java">
        <!-- enableSubPackages:是否让schema作为包的后缀 -->
        <property name="enableSubPackages" value="false" />
        <!-- 从数据库返回的值被清理前后的空格 -->
        <property name="trimStrings" value="true" />
    </javaModelGenerator>
    <!-- targetProject:mapper映射文件生成的位置 -->
    <sqlMapGenerator targetPackage="cn.xixipro.mapper" 
        targetProject="./src/main/java">
        <!-- enableSubPackages:是否让schema作为包的后缀 -->
        <property name="enableSubPackages" value="false" />
    </sqlMapGenerator>
    <!-- targetPackage：mapper接口生成的位置 -->
    <javaClientGenerator type="XMLMAPPER"
        targetPackage="cn.xixipro.mapper" 
        targetProject="../src/main/java">
        <!-- enableSubPackages:是否让schema作为包的后缀 -->
        <property name="enableSubPackages" value="false" />
    </javaClientGenerator>
    <!-- 指定数据库表 -->
    <table schema="" tableName="user"></table>
    <!-- <table schema="" tableName="tb_area"></table> -->

   </context>
 </generatorConfiguration>

 4.编写一个含有main方法的Java类，运行就会生成文件
    public class GeneratorSqlmap {

public void generator() throws Exception{

    List<String> warnings = new ArrayList<String>();
    boolean overwrite = true;
    //指定 逆向工程配置文件
    File configFile = new File("src/main/resources/generatorConfig.xml"); 
    ConfigurationParser cp = new ConfigurationParser(warnings);
    Configuration config = cp.parseConfiguration(configFile);
    DefaultShellCallback callback = new DefaultShellCallback(overwrite);
    MyBatisGenerator myBatisGenerator = new MyBatisGenerator(config,
            callback, warnings);
    myBatisGenerator.generate(null);

} 
public static void main(String[] args) throws Exception {
    try {
        GeneratorSqlmap generatorSqlmap = new GeneratorSqlmap();
        generatorSqlmap.generator();
    } catch (Exception e) {
        e.printStackTrace();
    }

  }
}

ps:在设置配置文件时，地址路径 Mac使用./ windows使用.\ 不然控制台没有报错但是生成文件找不到。
